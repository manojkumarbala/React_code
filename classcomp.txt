import React, { Component } from "react";
import axios from "axios";
import './Filter.css';
import Form from "./Form";
//added in branch1 and added again
//added here need to pull
class Filters extends Component {
  constructor(props) {
    super(props);
    this.state = {
      locname: "",
      result: [],
      selectedloc: "",
      suggestionList: false,
      flag: 0,
    };
  }

  handlefilter = async (value) => {
    const locationlist = await axios.get('http://localhost:8080/getDetails');
    console.log(locationlist);
    const locations = locationlist.data;

    const filteredLocations = locations.filter(item =>
      value && item.location && item.location.toLowerCase().includes(value.toLowerCase())
    );
    this.setState({
      result: filteredLocations,
      locname: value,
      suggestionList: true,
    });
  };

  handleAll = (value) => {
    this.setState({
      selectedloc: value,
      locname: value,
      result: [],
      suggestionList: false,
    });
  };

  handleFlag = () => {
    this.setState({
      flag: 1,
    });
  };

  resetToDefault = () => {
    this.setState({
      flag: 0,
    });
  };

  render() {
    const { locname, result, suggestionList, flag } = this.state;

    return (
      <div className="Form1">
        {
          flag === 0 && <div>
            <form className="SearchInputs">
              <input
                type="text"
                placeholder="Type Location"
                className="textForm"
                value={locname}
                onChange={(e) => this.handlefilter(e.target.value)}
              />
              <i className="fa fa-search"></i>
            </form>

            <div className="dataResult">
              {
                suggestionList &&
                (
                  <ul>
                    {
                      result.map((location, id) => {
                        return (
                          <li key={id} onClick={() => this.handleAll(location.location)}>
                            {location.location}
                          </li>
                        );
                      })
                    }
                  </ul>
                )
              }
            </div>
            <a href="#" onClick={this.handleFlag}>Not found Location.. Click here</a>
          </div>
        }

        {flag === 1 && <Form />}
      </div>
    );
  }
}

export default Filters;
